Composite Components

  • Verfügbar seit JSF 2.0
  • Komponenten als XHTML Dokumente
  • (Meist) ohne Java Code
  • Basieren auf dem JSF Resourcen Konzept

Resource Components


<h:outputStylesheet library="bootstrap" 
                       name="css/bootstrap.css" />

Markup


<link type="text/css" rel="stylesheet" 
  href="/demo/javax.faces.resource/css/bootstrap.css.jsf?ln=bootstrap"/>

Web Application Root

Composite Components


<!DOCTYPE html>
<html ...
    xmlns:cc="http://xmlns.jcp.org/jsf/composite" 
    ...>

  <cc:interface>
    <!-- Metadaten der Komponente -->
  </cc:interface>
  
  <cc:implementation>
    <!-- Implementierung der Komponente -->
  </cc:implementation>

</html>
          

<!DOCTYPE html>
<html ...
  xmlns:my="http://xmlns.jcp.org/jsf/composite/mylib" 
  ...>

  <!-- ... -->

  <my:foobar ... />

  <!-- ... -->

</html>

JavaScript in 3 Folien

Variablen


      // strings
      var name = 'Christian';

      // numbers
      var count = 5;
      
      // booleans
      var finished = true;
      
      // objects
      var conference = {
        name: 'WJAX',
        location: 'München',
        year: 2014
      };

Funktionen



      function sayHello(name) {
        alert("Hello" + name + "!");
      }          

      var sayHello = function(name) {
        alert("Hello" + name + "!");
      }          
      
      var object = {
        name: 'Greeter',
        sayHello: function(name) {
          alert("Hello" + name + "!");
        }          
      };
